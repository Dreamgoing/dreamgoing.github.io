enum EventType {
    READ = 0
};
​
class Epoll {
    // 封装了epoll I/O 多路复用的机制, Event demultiplexer
public:
    static const int NO_FLAGS = 0;
    static const int BLOCK_INDEFINITELY = -1;
    static const int MAX_EVENTS = 10;
​
    int socketfd;
    int fileDescriptor;
    struct epoll_event event;
    std::array<epoll_event, MAX_EVENTS> events{};
​
    Epoll() {
        port = "9000";
        socketfd = create_and_bind();
        make_socket_nonblocking(socketfd);
        if (listen(socketfd, SOMAXCONN) == -1) {
            std::cerr << "[ERROR] listen failed\n";
        }
        fileDescriptor = epoll_create1(NO_FLAGS);
        event.data.fd = socketfd;
        // 设置epoll event 为EPOLLIN(对应文件描述符可读)， EPOLLPRI(对应文件描述符有紧急事件可读)
        event.events = EPOLLIN | EPOLLET;
​
    }
​
    int wait() {
        return epoll_wait(fileDescriptor, events.data(), MAX_EVENTS, BLOCK_INDEFINITELY);
    }
​
    int control() {
        return epoll_ctl(fileDescriptor, EPOLL_CTL_ADD, socketfd, &event);
    }
​
    ~Epoll() {
        close(fileDescriptor);
    }
​
private:
​
    std::string port;
​
​
    int create_and_bind() {
        // 创建绑定具体某个端口的文件描述符
​
        struct addrinfo hints;
        memset(&hints, 0, sizeof(hints));
        hints.ai_family = AF_UNSPEC; // 返回IPv4 和 IPv6的选择
        hints.ai_socktype = SOCK_STREAM; // TCP类型
        hints.ai_flags = AI_PASSIVE; //fill in my IP for me
​
        struct addrinfo *result;
        //getaddrinfo 将主机名， 主机地址，端口转换成可用的socket链表
        //getaddrinfo 返回可用的socket链表，存储在result中
        int sockt = getaddrinfo(nullptr, port.c_str(), &hints, &result);
​
        if (sockt != 0) {
            std::cerr << "[ERROR] getaddrinfo error" << std::endl;
        }
​
        struct addrinfo *rp;
        int socketfd;
        for (rp = result; rp != nullptr; rp = rp->ai_next) {
            // socket 获取socket的文件描述符
            socketfd = socket(rp->ai_family, rp->ai_socktype, rp->ai_protocol);
            if (socketfd == -1) {
                continue;
            }
​
            // bind 将socket绑定到具体的某个端口上
            sockt = bind(socketfd, rp->ai_addr, rp->ai_addrlen);
            if (sockt == 0) {
                break;
            }
            // 发生错误则继续找下一个可用的socket
            close(socketfd);
​
        }
        if (rp == nullptr) {
            std::cerr << "[ERROR] bind error" << std::endl;
        }
        freeaddrinfo(result);
​
        return socketfd;
    }
​
    bool make_socket_nonblocking(int socketfd) {
        // fcntl 修改一个文件描述符,
        int flags = fcntl(socketfd, F_GETFL, 0);
        if (flags == -1) {
            std::cerr << "[ERROR] fcntl failed (F_GETFL)\n";
            return false;
        }
​
        flags |= O_NONBLOCK;
        int s = fcntl(socketfd, F_SETFL, flags);
        if (s == -1) {
            std::cerr << "[ERROR] fcntl failed (F_SETFL)\n";
            return false;
        }
        return true;
    }
​
public:
    bool accept_connection(int socketfd, struct epoll_event &event, int epollfd) {
        struct sockaddr in_addr;
        socklen_t in_len = sizeof(in_addr);
        int infd = accept(socketfd, &in_addr, &in_len);
        if (infd == -1) {
            // EAGAIN， EWOULDBLOCK 表示系统调用失败，即现在没有可用的数据，请稍后重试
            if (errno == EAGAIN || errno == EWOULDBLOCK) { // Done processing incoming connections
                return false;
            } else {
                std::cerr << "[ERROR] accept failed\n";
                return false;
            }
        }
​
        std::string hbuf(NI_MAXHOST, '\0');
        std::string sbuf(NI_MAXSERV, '\0');
​
        //getnameinfo 与 getaddrinfo相反，即将socket地址结果，转化成对应的主机名，和服务名，端口
        if (getnameinfo(&in_addr, in_len,
                        const_cast<char *>(hbuf.data()), hbuf.size(),
                        const_cast<char *>(sbuf.data()), sbuf.size(),
                        NI_NUMERICHOST | NI_NUMERICSERV) == 0) {
            std::cout << "[INFO] Accepted connection on descriptor " << infd << "(host=" << hbuf << ", port=" << sbuf
                      << ")" << "\n";
        }
​
        if (!make_socket_nonblocking(infd)) {
            std::cerr << "[ERROR] make_socket_nonblocking failed\n";
            return false;
        }
​
        event.data.fd = infd;
        // 按位与运算，设置事件 为EPOLLIN可读状态, EPOLLET边沿触发
        event.events = EPOLLIN | EPOLLET;
​
        // 将新建连接的文件描述符，添加到epoll实例中， EPOLL_CTL_ADD 即为添加
        if (epoll_ctl(epollfd, EPOLL_CTL_ADD, infd, &event) == -1) {
            std::cerr << "[ERROR] epoll_ctl failed\n";
            return false;
        }
​
        return true;
​
    }
​
};

class EventHandler {
    // Event Handler
​
public:
    int handle_event(EventType et, int fd) {
        if (et == READ) {
            return read_data(fd);
        }
        return 0;
    }
​
    // 对于网络编程，或者socket编程来说，需要使用如下的I/O方式，RIO的方式，更加健壮，且避免了频繁从内核态，向用户态的切换
    bool read_data(int fd) {
        // 类似于RIO（Robust I/O）的一种思想，带缓冲的输入函数
        char buf[512];
​
        //read 系统调用，内核态
        auto count = read(fd, buf, 512);
        if (count == -1) {
            if (errno == EAGAIN) { // read all data
                return false;
            }
        } else if (count == 0) { // EOF - remote closed connection
            std::cout << "[INFO] Close " << fd << "\n";
            close(fd);
            return false;
        }
​
        std::cout << fd << " says: " << buf;
​
        return true;
    }
​
    bool wirte_data(int fd) {
        // 类似于RIO，带缓冲的输出函数
        // write HTTP 200
        char buf[512];
        sprintf(buf, "HTTP/1.1 200 OK\r\nContent-Length: %d\r\n\r\nHello World!", 12);
        int nwrite, data_size = strlen(buf);
        int n = data_size;
        while (n > 0) {
            // write 系统调用，内核态
            nwrite = write(fd, buf + data_size - n, n);
            if (nwrite < n) {
                if (nwrite == -1 && errno != EAGAIN) {
                    std::cerr << "[ERROR] write data error" << std::endl;
                }
                break;
            }
            n -= nwrite;
        }
    }
​
};

class Reactor {
    // Dispatcher
public:
    Reactor() {
        epoll.control();
    }
​
    //注册对应的回调函数到handlers中
    void addHandler(std::string event, EventHandler callback) {
        handlers.emplace(std::move(event), callback);
    }
​
    void run() {
​
        while (true) {
            int numberOfEvents = wait();
            for (int i = 0; i < numberOfEvents; ++i) {
                if (epoll.socketfd == epoll.events[i].data.fd) {
                    // 新建的一个连接
                    std::cout << "New connection" << epoll.events[i].data.fd << std::endl;
                    while (epoll.accept_connection(epoll.socketfd, epoll.event, epoll.fileDescriptor)) {};
                    continue;
                }
                // 按位与 & 运算，处理包含关系，具体可参考EPOLL宏定义
                if (epoll.events[i].events & EPOLLIN) {
​
                    // EPOLLIN 表示对应的文件描述符可读
                    auto fd = epoll.events[i].data.fd;
​
                    // 进行事件的处理， 读取事件并进行相应处理, 这里为读出所有请求的数据
                    while (EventHandler().read_data(fd)) {}
​
                    // 所有读事件处理完毕，将event的状态置为 EPOLLOUT: 表示对应文件描述符可写
                    epoll.event.data.fd = fd;
                    epoll.event.events = epoll.events[i].events | EPOLLOUT;
​
                    // 修改epoll监听的对应实例描述符为可读状态
                    epoll_ctl(epoll.fileDescriptor, EPOLL_CTL_MOD, fd, &epoll.event);
                }
​
                if (epoll.events[i].events & EPOLLOUT) {
                    // EPOLLOUT 表示对应文件描述符可写
                    auto fd = epoll.events[i].data.fd;
                    EventHandler().wirte_data(fd);
​
                }
            }
        }
    }
​
​
private:
    // handlers Table, 存储事件以及其对应的handlers
    std::unordered_map<std::string, EventHandler> handlers{};
    Epoll epoll;
​
    int wait() {
        int numberOfEvents = epoll.wait();
        return numberOfEvents;
    }
};
​
int main() {
    Reactor reactor;
    reactor.run();
}
